### Task 1: Headless Launch — No Terminal Window

**Files:**

- Modify: `VoiceInput-Mac.command`

The macOS `.command` file currently opens Terminal, launches in background, then tries to close the Terminal window with osascript (unreliable, user sees flash). Fix: `start.py` re-execs itself with `pythonw`-equivalent to suppress terminal entirely.

- [ ] **Step 1: Modify `start.py` to re-exec headless**

At the top of the `if __name__ == "__main__"` block, before `ensure_deps()`, add logic that detects if running in a visible terminal and re-execs itself detached:

```python
if __name__ == "__main__":
    # Re-launch headless if running from a terminal
    if sys.platform == "darwin" and sys.stdout and sys.stdout.isatty():
        import subprocess as _sp
        _sp.Popen(
            [sys.executable, __file__],
            stdout=open(os.devnull, "w"),
            stderr=open(os.devnull, "w"),
            stdin=open(os.devnull, "r"),
            start_new_session=True,
        )
        sys.exit(0)
    elif sys.platform == "win32":
        # On Windows, check if launched from pythonw (no console)
        # If running with python.exe (has console), re-launch with pythonw.exe
        exe = sys.executable.lower()
        if "pythonw" not in exe and sys.stdout and hasattr(sys.stdout, "isatty") and sys.stdout.isatty():
            pythonw = exe.replace("python.exe", "pythonw.exe")
            if os.path.exists(pythonw):
                import subprocess as _sp
                _sp.Popen(
                    [pythonw, __file__],
                    stdout=open(os.devnull, "w"),
                    stderr=open(os.devnull, "w"),
                    stdin=open(os.devnull, "r"),
                    creationflags=0x00000008,  # DETACHED_PROCESS
                )
                sys.exit(0)

    # Ensure stdout/stderr exist (pythonw sets them to None)
    if sys.stdout is None:
        sys.stdout = io.StringIO()
    if sys.stderr is None:
        sys.stderr = open(os.devnull, "w")

    ensure_deps()
    from voice_input.app import main
    main()
```

```python
class _MacHotkeyListener:
    """Listen for configurable modifier key on macOS using Quartz CGEventTap."""

    # Map config key string -> (keycode, flag_bit)
    _KEY_MAP = {
        "fn":          (63, 0x800000),    # NSEventModifierFlagSecondaryFn
        "Key.alt_r":   (61, 0x40),        # Right Option, kCGEventFlagMaskAlternate sub-bit
        "Key.cmd_r":   (54, 0x10),        # Right Command, kCGEventFlagMaskCommand sub-bit
        "Key.ctrl_l":  (59, 0x1),         # Left Control, kCGEventFlagMaskControl sub-bit
    }

    def __init__(self, hotkey_str, on_press, on_release):
        self._hotkey_str = hotkey_str
        self._on_press = on_press
        self._on_release = on_release
        self._key_down = False
        self._thread = None
        self._tap = None
        self._loop_ref = None
        self._keycode, self._flag = self._KEY_MAP.get(hotkey_str, (63, 0x800000))

    def start(self) -> None:
        self._thread = threading.Thread(target=self._run, daemon=True)
        self._thread.start()

    def _run(self) -> None:
        import Quartz
        from Foundation import NSRunLoop, NSDate

        keycode = self._keycode
        flag = self._flag

        def callback(proxy, event_type, event, refcon):
            kc = Quartz.CGEventGetIntegerValueField(event, Quartz.kCGKeyboardEventKeycode)
            flags = Quartz.CGEventGetFlags(event)

            if kc == keycode:
                pressed = bool(flags & flag)
                if pressed and not self._key_down:
                    self._key_down = True
                    try:
                        self._on_press()
                    except Exception:
                        log.exception("Error in on_press")
                elif not pressed and self._key_down:
                    self._key_down = False
                    try:
                        self._on_release()
                    except Exception:
                        log.exception("Error in on_release")
            return event

        event_mask = Quartz.CGEventMaskBit(Quartz.kCGEventFlagsChanged)
        self._tap = Quartz.CGEventTapCreate(
            Quartz.kCGSessionEventTap,
            Quartz.kCGHeadInsertEventTap,
            Quartz.kCGEventTapOptionListenOnly,
            event_mask,
            callback,
            None,
        )

        if self._tap is None:
            log.error("Failed to create event tap. Grant Accessibility permissions.")
            return

        source = Quartz.CFMachPortCreateRunLoopSource(None, self._tap, 0)
        self._loop_ref = Quartz.CFRunLoopGetCurrent()
        Quartz.CFRunLoopAddSource(self._loop_ref, source, Quartz.kCFRunLoopDefaultMode)
        Quartz.CGEventTapEnable(self._tap, True)
        Quartz.CFRunLoopRun()

    def stop(self) -> None:
        self._key_down = False
        if self._loop_ref:
            import Quartz
            Quartz.CFRunLoopStop(self._loop_ref)
            self._loop_ref = None

    def update_hotkey(self, hotkey_str):
        """Update requires restart of listener since CGEventTap callback captures keycode."""
        self._hotkey_str = hotkey_str
        self._keycode, self._flag = self._KEY_MAP.get(hotkey_str, (63, 0x800000))
```

When a new model is selected and needs downloading, faster-whisper downloads it on first `WhisperModel(...)` call. We hook into this by writing progress to the status file, which the control panel already polls every 1.2s.

- [ ] **Step 1: Add download progress writing to `transcriber.py`**

The `_load_model` function should write status before loading (since faster-whisper downloads during `WhisperModel()` init):

```python
def _load_model(model_name: str, use_gpu: bool):
    from faster_whisper import WhisperModel

    device = "cuda" if use_gpu else "cpu"
    compute_type = "float16" if use_gpu else "int8"

    # Check if model needs downloading
    downloaded = False
    if config.MODEL_DIR.exists():
        for entry in config.MODEL_DIR.iterdir():
            if entry.is_dir() and model_name in entry.name:
                downloaded = True
                break

    if not downloaded:
        log.info("Downloading model %s (this may take a while)...", model_name)
        _write_download_status(model_name, "downloading")
    else:
        log.info("Loading model %s on %s...", model_name, device)
        _write_download_status(model_name, "loading")

    model = WhisperModel(
        model_name,
        device=device,
        compute_type=compute_type,
        download_root=str(config.MODEL_DIR),
    )

    _write_download_status(model_name, "ready")
    return model


def _write_download_status(model_name: str, state: str):
    """Write model loading status so control panel can show it."""
    import json
    from datetime import datetime
    try:
        status = {}
        if config.STATUS_PATH.exists():
            status = json.loads(config.STATUS_PATH.read_text(encoding="utf-8"))
        if state == "downloading":
            status["state"] = "downloading"
            status["detail"] = f"Downloading model: {model_name}..."
        elif state == "loading":
            status["state"] = "loading"
            status["detail"] = f"Loading model: {model_name}..."
        elif state == "ready":
            status["state"] = "hidden"
            status["detail"] = "Ready"
        status["updated_at"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        config.STATUS_PATH.write_text(
            json.dumps(status, ensure_ascii=False, indent=2),
            encoding="utf-8",
        )
    except Exception:
        pass
```
